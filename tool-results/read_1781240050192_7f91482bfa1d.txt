   183→function formatDate(dateStr: string, range: RangeOption): string {
   184→  const d = new Date(dateStr)
   185→  if (range === '1D') return d.toLocaleTimeString('en-IN', { hour: '2-digit', minute: '2-digit', hour12: false })
   186→  if (range === '1W') return d.toLocaleDateString('en-IN', { weekday: 'short', hour: '2-digit', minute: '2-digit', hour12: false })
   187→  if (range === '1M' || range === '3M') return d.toLocaleDateString('en-IN', { month: 'short', day: 'numeric' })
   188→  return d.toLocaleDateString('en-IN', { month: 'short', year: '2-digit' })
   189→}
   190→
   191→function formatExpiry(dateStr: string): string {
   192→  if (!dateStr) return '--'
   193→  const d = new Date(dateStr)
   194→  return d.toLocaleDateString('en-IN', { day: '2-digit', month: 'short', year: 'numeric' })
   195→}

   212→function CustomTooltip({ active, payload, range }: { active?: boolean; payload?: Array<{ payload: CandleData }>; range: RangeOption }) {
   213→  if (!active || !payload || !payload.length) return null
   214→  const d = payload[0].payload
   215→  const isUp = d.close >= d.open
   216→
   217→  return (
   218→    <div className="bg-white border border-[#e5e7eb] rounded-lg p-3 shadow-xl text-xs">
   219→      <div className="font-semibold text-[#1a1a1a] mb-1.5">{formatDate(d.date, range)}</div>
   220→      <div className="grid grid-cols-2 gap-x-4 gap-y-0.5">
   221→        <span className="text-[#6b7280]">Open</span>
   222→        <span className="font-mono font-tabular text-right">{d.open.toLocaleString('en-IN', { maximumFractionDigits: 2 })}</span>
   223→        <span className="text-[#6b7280]">High</span>
   224→        <span className="font-mono font-tabular text-right">{d.high.toLocaleString('en-IN', { maximumFractionDigits: 2 })}</span>
   225→        <span className="text-[#6b7280]">Low</span>
   226→        <span className="font-mono font-tabular text-right">{d.low.toLocaleString('en-IN', { maximumFractionDigits: 2 })}</span>
   227→        <span className="text-[#6b7280]">Close</span>
   228→        <span className={`font-mono font-tabular text-right font-semibold ${isUp ? 'text-[#00B386]' : 'text-[#EB5B3C]'}`}>
   229→          {d.close.toLocaleString('en-IN', { maximumFractionDigits: 2 })}
   230→        </span>
   231→        {d.volume > 0 && (
   232→          <>
   233→            <span className="text-[#6b7280]">Volume</span>
   234→            <span className="font-mono font-tabular text-right">{formatVolume(d.volume)}</span>
   235→          </>
   236→        )}
   237→      </div>
   238→    </div>
   239→  )
   240→}

   242→// ─── Range Bar Component (Groww-style) ────────────────────────────────────

   244→function RangeBar({ label, low, high, current, lowLabel, highLabel }: {
   245→  label: string
   246→  low: number
   247→  high: number
   248→  current: number
   249→  lowLabel?: string
   250→  highLabel?: string
   251→}) {
   252→  const range = high - low
   253→  const position = range > 0 ? Math.min(100, Math.max(0, ((current - low) / range) * 100)) : 50
   254→
   255→  return (
   256→    <div className="space-y-2">
   257→      <div className="flex items-center justify-between">
   258→        <span className="text-xs text-[#6b7280]">{lowLabel || label + ' Low'}</span>
   259→        <span className="text-xs text-[#6b7280]">{highLabel || label + ' High'}</span>
   260→      </div>
   261→      <div className="flex items-center justify-between text-sm font-mono font-tabular font-semibold">
   262→        <span className="text-[#EB5B3C]">{low.toLocaleString('en-IN', { maximumFractionDigits: 2 })}</span>
   263→        <span className="text-[#00B386]">{high.toLocaleString('en-IN', { maximumFractionDigits: 2 })}</span>
   264→      </div>
   265→      <div className="h-2 rounded-full bg-[#f0f2f5] relative overflow-hidden">
   266→        <div
   267→          className="absolute top-0 left-0 h-full rounded-full"
   268→          style={{
   269→            width: '100%',
   270→            background: 'linear-gradient(to right, #eb5b3c, #00d09c)',
   271→            opacity: 0.25,
   272→          }}
   273→        />
   274→        <div
   275→          className="absolute top-1/2 -translate-y-1/2 size-3 bg-[#00d09c] border-2 border-white rounded-full shadow-sm z-10"
   276→          style={{ left: `calc(${position}% - 6px)` }}
   277→        />
   278→      </div>
   279→    </div>
   280→  )
   281→}

   339→export function StockOverviewPage() {

   343→

   346→
   347→  // State

   361→
   362→  // Watchlist state - use shared store for instant sync
   363→  const { isInWatchlist: isInWatchlistStore, addSymbol, removeSymbol, setSymbols: setWatchlistSymbols, loaded: watchlistLoaded } = useWatchlistStore()
   364→  const isInWatchlist = isInWatchlistStore(selectedStockSymbol || '')
   365→  const [watchlistLoading, setWatchlistLoading] = useState(false)
   366→
   367→  // Trade modal state
   368→  const [showTradePanel, setShowTradePanel] = useState(false)
   369→  const [orderSide, setOrderSide] = useState<'buy' | 'sell'>('buy')
   370→  const [orderType, setOrderType] = useState('MARKET')
   371→  const [productType, setProductType] = useState('INTRADAY')
   372→  const [quantity, setQuantity] = useState(10)
   373→  const [price, setPrice] = useState('')

   376→  const [confirmModalOpen, setConfirmModalOpen] = useState(false)
   377→  const [confirmData, setConfirmData] = useState<TradeConfirmData | null>(null)
   378→
   379→  // ─── Load watchlist into shared store ──────────────────────────
   380→  useEffect(() => {
   381→    if (!token || watchlistLoaded) return
   382→    fetch('/api/trade/watchlist', {
   383→      headers: { Authorization: `Bearer ${token}` },
   384→    })
   385→      .then(res => res.json())
   386→      .then(data => {

   388→          setWatchlistSymbols(data.data.map((item: { symbol: string }) => item.symbol))

   395→  const toggleWatchlist = async () => {

   402→          headers: {
   403→            Authorization: `Bearer ${token}`,
   404→            'Content-Type': 'application/json',

   406→          body: JSON.stringify({ symbol: stockDetail.symbol }),

   410→          removeSymbol(stockDetail.symbol)

   412→        } else {
   413→          toast.error(data.error || 'Failed to remove')

   416→        const res = await fetch('/api/trade/watchlist', {
   417→          method: 'POST',
   418→          headers: {
   419→            Authorization: `Bearer ${token}`,
   420→            'Content-Type': 'application/json',
   421→          },
   422→          body: JSON.stringify({ symbol: stockDetail.symbol }),

   428→        } else {
   429→          toast.error(data.error || 'Failed to add')
   430→        }
   431→      }
   432→    } catch {
   433→      toast.error('Network error')
   434→    } finally {
   435→      setWatchlistLoading(false)
   436→    }
   437→  }
   438→
   439→  // ─── Fetch stock detail ─────────────────────────────────────────
   440→  const fetchDetail = useCallback(async (isBackgroundRefresh = false) => {

   444→      const res = await fetch(`/api/stocks/detail/${selectedStockSymbol}`)

   451→      }
   452→    } catch {
   453→      // Keep previous data
   454→    } finally {
   455→      if (!isBackgroundRefresh) setDetailLoading(false)

   458→

   460→  const fetchChart = useCallback(async () => {
   461→    if (!selectedStockSymbol) return

   477→  // ─── Fetch F&O data ─────────────────────────────────────────────
   478→  const fetchFnoData = useCallback(async () => {

   481→    try {
   482→      const res = await fetch(`/api/stocks/fno/${selectedStockSymbol}`)
   483→      if (res.ok) {
   484→        const json = await res.json()
   485→        if (json.success && json.data) {
   486→          setFnoData(json.data)
   487→          if (json.data.optionChainSummary?.nearestExpiry && !selectedExpiry) {
   488→            setSelectedExpiry(json.data.optionChainSummary.nearestExpiry)
   489→          }
   490→        }
   491→      }
   492→    } catch {
   493→      // Keep previous data
   494→    } finally {
   495→      setFnoLoading(false)
   496→    }
   497→  }, [selectedStockSymbol, selectedExpiry])
   498→
   499→  useEffect(() => {
   500→    fetchDetail()
   501→  }, [fetchDetail])
   502→
   503→  useEffect(() => {
   504→    fetchChart()
   505→  }, [fetchChart])
   506→

   511→
   512→    const previousClose = wsQuote.ohlc.close - wsQuote.net_change

   514→
   515→    setStockDetail(prev => {
   516→      if (!prev) return prev
   517→      return {
   518→        ...prev,
   519→        currentPrice: wsQuote.last_price,
   520→        change: wsQuote.net_change,
   521→        changePercent,
   522→        open: wsQuote.ohlc.open || prev.open,
   523→        high: wsQuote.ohlc.high || prev.high,
   524→        low: wsQuote.ohlc.low || prev.low,
   525→        close: wsQuote.ohlc.close || prev.close,
   526→        previousClose,
   527→        volume: wsQuote.volume || prev.volume,
   528→      }
   529→    })
   530→  }, [wsQuote])
   531→
   532→  // Fallback: REST polling only when WebSocket is disconnected
   533→  useEffect(() => {
   534→    if (wsStatus === 'connected') return // No need for REST polling when WS is active
   535→    const interval = setInterval(() => fetchDetail(true), 1000) // 1s for near-real-time updates
   536→    return () => clearInterval(interval)
   537→  }, [fetchDetail, wsStatus])
   538→
   539→  // Fetch F&O data when F&O tab is active or when stock has F&O
   540→  useEffect(() => {
   541→    if (stockDetail && (stockDetail.isFuturesAvailable || stockDetail.isOptionsAvailable)) {
   542→      fetchFnoData()
   543→    }
   544→  }, [stockDetail?.isFuturesAvailable, stockDetail?.isOptionsAvailable, fetchFnoData])

   550→    return () => clearInterval(interval)
   551→  }, [overviewTab, fetchFnoData])
   552→
   553→  // Set price when stock changes
   554→  useEffect(() => {
   555→    if (stockDetail) {
